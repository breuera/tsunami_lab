312  template<typename InputIterator, typename Sentinel = InputIterator>
313  std::string rangeToString(InputIterator first, Sentinel last) {
314  ReusableStringStream rss;
315  rss << "{ ";
316  if (first != last) {
317  rss << ::Catch::Detail::stringify(*first);
318  for (++first; first != last; ++first)
319  rss << ", " << ::Catch::Detail::stringify(*first);
320  }
321  rss << " }";
322  return rss.str();
323  }
